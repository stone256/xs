Start _standard.tpl.html

standard type 1

{{$bbb}} → variable $TEMPLATE_DATA[bbb]. *you can omit $ , just use {{bbb}} when in the first positionbbb
{{ccc}} → variable is array, $TEMPLATE_DATA[ccc] = ['c1','c2','c3',5=>'bbb'] c1,c2,c3,bbb
{{ccc,5}} → array element, $TEMPLATE_DATA[ccc] = ['c1','c2','c3',5=>'bbb'], so $TEMPLATE_DATA[ccc][5] = bbb bbb
{{abcd:this is default value}} → default valuethis is default value
{{abcd:$n,4}} → use variable $TEMPLATE_DATA[n][4] as default4.4
{{abcd:$n,m}} → use variable $TEMPLATE_DATA[n][m] as default1,2,3,4,this is 5,a
{{*dddd*}} → comments{{dddd}}

standard type 2

{{bbb==bbb?bbb is eq bbb:bbb is not eq bbb}} → if $TEMPLATE_DATA[bbb] == 'bbb' bbb is eq bbb
{{ddd?has ddd: dddd is not existed |strtoupper}} → if($TEMPLATE_DATA[ddd]) and apply function strtoupper to result DDDD IS NOT EXISTED
{{$n,m,2>5?gt 5:$n,0}} → if($TEMPLATE_DATA[n][m][2]>5) 'gt 5' else $TEMPLATE_DATA[n][0] 111
{{$n,m,2<5?lt 5:$n,0}} → if($TEMPLATE_DATA[n][m][2]<5) 'lt 5' else $TEMPLATE_DATA[n][0] lt 5

standard type 3

{{$n,m,0#000:111:222:333:5=555:a=$bbb}} → [000,111,222,333,5=>555,a=>$bbb][$TEMPLATE_DATA[n][m][0]] → [000,111,222,333,5=555,a=$bbb][1] as $TEMPLATE_DATA[n][m][0]=1111
{{$n,m,2#000:111:222:333:5=555:a=$bbb}} → [000,111,222,333,5=>555,a=>$bbb][$TEMPLATE_DATA[$TEMPLATE_DATA[n]][m][2]] → [000,111,222,333,5=555,a=>$bbb][3]333
{{$n,m,22#000:111:222:333:5=555:a=$ccc,2}} → since $TEMPLATE_DATA[n][m][22] = 'a', $TEMPLATE_DATA[ccc] = ['c1','c2','c3',5=>'bbb'] this translate to [000,111,222,333,5=>555,a=>$ccc,2][a][2] = c3 c3
{{$n,5#$n,m}} → $TEMPLATE_DATA[n][5] = 5, $TEMPLATE_DATA[n][m] = [1,2,3,4, 5=>'this is 5', 22=>'a'] this translate to [1,2,3,4, 5=>'this is 5', 22=>'a'][5] this is 5
